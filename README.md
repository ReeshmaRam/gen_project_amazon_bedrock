# gen_project_amazon_bedrock
#How to run 
### CREATE ENVIRONMENT
''' bash 
conda create -n llmapp python=3.8 -y 
'''
### ACTIVATE  ENVIRONMENT
''' bash 
conda activate llmapp 
'''
### INSTALL THE REQUIREMENTS PACKAGE
''' bash 
pip install -r requirements.txt
'''
### RUN YOUR APP
''' bash 
streamlit run main.py
'''
